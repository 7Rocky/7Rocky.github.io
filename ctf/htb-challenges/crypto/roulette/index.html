import os
import random

N = 128
M = 30
b = 32
MAGIC = 0xb249b015
mask = (1 << b) - 1

class Twister:
    def rol(self, x, d):
        ans = (x << d) & mask
        ans = ans | (x >> (b - d))
        return ans & mask

    def __init__(self, state=None):
        self.index = N
        if state is not None:
            assert len(state) == N
            self.STATE = state[:]
        else:
            self.STATE = [0] * N
            for i in range(N):
                self.STATE[i] = int(os.urandom(4).encode('hex'), 16)  
            for i in range(N):
                self.STATE[i] ^= random.getrandbits(32)

    def twist(self):
        for i in range(N):
            self.STATE[i] ^= self.rol(self.STATE[(i+1) % N], 3)
            self.STATE[i] ^= self.rol(self.STATE[(i+M) % N], b - 9)
            self.STATE[i] ^= MAGIC

    def rand(self):
        if self.index >= N:
            self.twist()
            self.index = 0
        y = self.STATE[self.index]
        y ^= self.rol(y, 7)
        y ^= self.rol(y, b - 15)
        self.index += 1

        return y & mask
