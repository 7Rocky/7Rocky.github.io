func initBoard(seed int64, size, mines int) [][]Cell {
        r := rand.New(rand.NewSource(seed))
        board := make([][]Cell, size)
        numMines := 0

        for i := range size {
                board[i] = make([]Cell, size)
        }

        for numMines < mines {
                a := r.Intn(size)
                b := r.Intn(size)

                if !board[a][b].Bomb {
                        board[a][b].Bomb = true

                        for k := -1; k <= 1; k++ {
                                for m := -1; m <= 1; m++ {
                                        if k+a < size && m+b < size && k+a >= 0 && m+b >= 0 {  
                                                board[k+a][m+b].Adjacent++
                                        }
                                }
                        }

                        numMines++
                }
        }

        return board
}
